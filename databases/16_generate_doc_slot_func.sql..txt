CREATE OR REPLACE FUNCTION generate_available_slots(
    p_doctor_id bigint,
    p_date date
) 
RETURNS TABLE (slot_start time, slot_end time) 
LANGUAGE plpgsql
AS $$
DECLARE
    duration_minutes integer;
BEGIN
    -- 1. Security Check: Instantly return an empty grid if the doctor is on vacation
    IF EXISTS (
        SELECT 1 
        FROM doctor_unavailable_dates 
        WHERE doctor_id = p_doctor_id AND unavailable_date = p_date
    ) THEN
        RETURN; 
    END IF;

    -- 2. Fetch the doctor's custom duration (Defaults to 30 if null)
    SELECT COALESCE(consultation_duration, 30)
    INTO duration_minutes
    FROM doctors 
    WHERE id = p_doctor_id AND active = true;

    -- If doctor doesn't exist or is inactive, exit silently
    IF duration_minutes IS NULL THEN
        RETURN;
    END IF;

    -- 3. Search your existing rows directly and filter out already taken appointments
    RETURN QUERY
    SELECT 
        ds.appointment_time AS slot_start,
        (ds.appointment_time + (duration_minutes || ' minutes')::interval)::time AS slot_end
    FROM doctor_schedule ds
    WHERE ds.doctor_id = p_doctor_id 
      AND ds.appointment_date = p_date
      -- Security Lock: Only show slots that aren't already booked in the appointments table
      AND NOT EXISTS (
          SELECT 1 
          FROM appointments a 
          WHERE a.doctor_id = p_doctor_id 
            AND a.appointment_date = p_date
            AND a.appointment_time = ds.appointment_time
      )
    ORDER BY ds.appointment_time ASC;
END;
$$;
